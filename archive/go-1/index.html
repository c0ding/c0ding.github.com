Go 读取命令行参数的3种方式

2018-12-04
0
读取命令行
cli框架

779 words 4 minute read

对于初学者，想要尽快熟练掌握go语言，那就要大量的刻意练习，多以实践的方式为主。

今天为大家推荐的刻意练习是：编写一个简单的数学计算器，读取命令行参数进行运算。如此既可以熟练语言，也为开发命令行应用起到入门效果。

本篇使用3种方式来读取命令行：

  • 内置 os 包读取命令参数
  • 内置 flag 包读取命令参数
  • cli 框架读取命令参数 比较流行

题目分析

实现 “简单数学计算” 这个功能，读取命令行参数：比如 go run main.go add 1 2
除文件名之外的第一个元素：解析为进行数学运算符，比如： add、sub、mul、sqrt。
其余参数表示：进行操作的数值。

1. OS 获取命令行参数

os包基本使用

# 1.接受的参数，是一个切片类型
os.Args

# 2.将字符串数值转换为整型
strconv.Atoi  

# 3.整型转换为字符串
strconv.Itoa

# 4.将字符串数值转换为浮点型
strconv.ParseFloat

实现代码

注意参与运算前先过滤异常情况，如参数的合法性、类型转换

func CalcByOs() error {
    args := os.Args // 获取参数列表
    if len(args) < 3 || args == nil {  //过滤 异常情况，提示
        help()
        return nil
    }
    operate := args[1] // 获取运算符
    switch operate {
    case "add":{
            rt := 0
            number_one, err1 := strconv.Atoi(args[2]) // 获取数值
            number_two, err2 := strconv.Atoi(args[3])
            if err1 == nil && err2 == nil {	 
                rt = number_one + number_two // 两个数值做相应的运算
                fmt.Println("Result ", rt)	// 打印结果
            }
        }
    case "sub":
        {
            rt := 0
            number_one, err1 := strconv.Atoi(args[2])
            number_two, err2 := strconv.Atoi(args[3])
            if err1 == nil && err2 == nil {
                rt += number_one - number_two
                fmt.Println("Result ", rt)
            }
        }
    case "mul":
        {
            rt := 1
            number_one, err1 := strconv.Atoi(args[2])
            number_two, err2 := strconv.Atoi(args[3])
            if err1 == nil && err2 == nil {
                rt = number_one * number_two
                fmt.Println("Result ", rt)
            }
        }
    case "sqrt":
        {
            rt := float64(0)
            if len(args) != 3 {
                fmt.Println("Usage: sqrt 2, return 1.4142135623730951")
                return nil
            }
            number_one, err := strconv.ParseFloat(args[2], 64)
            if err == nil {
                rt = math.Sqrt(number_one)
                fmt.Println("Result ", rt)
            }
        }
    default:
        help() //错误提示

    }
    return nil
}

结果：

$ go run main.go add 1 2
Result  3

$ go run main.go sub 1 2
Result  -1

$ go run main.go mul 10 20
Result  200

$ go run main.go sqrt 2
Result  1.4142135623730951

2. flag 获取命令行参数

flag 包基本使用
1，定义变量
2，将变量与命令参数绑定
3，调用开始 解析命令
4，计算

实现代码

func CalcByFlag() error {
	var operation string
	var numOne float64
	var numTwo float64
	
	// 参数依次说明：绑定变量，命令参数名，命令参数默认值，命令参数说明
	flag.StringVar(&operation, "o", "add", "operation for this tool")
	flag.Float64Var(&numOne, "n1", 0,"The first number")
	flag.Float64Var(&numTwo, "n2", 0, "The second number")

	// 解析命令参数
	flag.Parse() 	
	fmt.Println(numOne, numTwo)
	if operation == "add" {
		rt := numOne + numTwo
		fmt.Println("Result ", rt)
	} else if operation == "sub" {
		rt := numOne - numTwo
		fmt.Println("Result ", rt)
	} else if operation == "mul" {
		rt := numOne * numTwo
		fmt.Println("Result ", rt)
	} else if operation == "sqrt" {
		rt := math.Sqrt(numOne)
		fmt.Println("Result ", rt)
	} else {
		help()
	}
	return nil
}

调用结果：

$ go run main.go -o add -n1 1 -n2 1.2
Result  2.2

$ go run main.go -o sub -n1 1 -n2 2
Result  -1

$ go run main.go -o mul -n1 1 -n2 20
Result  20

$ go run main.go -o sqrt -n1 2
Result  1.4142135623730951

3. CLI 框架

CLI 基本使用
1. 获取框架， $go get github.com/urfave/cli
2. 实例化一个 cli APP
3. 设置 cli APP 的名字、介绍
4. 设置 cli APP 的命令参数
5. 编写 cli APP 的功能
6. 运行监听 cli APP

简单使用

	app := cli.NewApp()
	app.Name = "cli框架"
	app.Usage = "简单使用"
	app.Version = "1.0"
	app.Action = func(c *cli.Context) error {
		fmt.Println("我的cli app 动能")
		return nil
	}

	app.Run(os.Args)

代码实现

    //第1步
    app := cli.NewApp() 
    //第2步
    app.Name = "cli 计算器"
    app.Usage = "基于 cli 框架的 命令行计算器"
    app.Version = "0.1.0"

    //第3步
    // 设置命令参数名，可设置多个，如：设置一个命令参数的 全称和简写
    // 设置默认值
    // 设置命令参数说明
    app.Flags = []cli.Flag{
        cli.StringFlag{
            Name:  "operation, o",
            Value: "add",
            Usage: "calc operation",
        },
        cli.Float64Flag{
            Name:  "numberone, n1",
            Value: 0,
            Usage: "number one for operation",
        },
        cli.Float64Flag{
            Name:  "numbertwo, n2",
            Value: 0,
            Usage: "number two for operation",
        },
    }
    //第4步
    app.Action = func(c *cli.Context) error {
        operation := c.String("operation")
        numberone := c.Float64("numberone")
        numbertwo := c.Float64("numbertwo")
        //fmt.Println(operation, numberone, numbertwo)
        if operation == "add" {
            rt := numberone + numbertwo
            fmt.Println("Result ", rt)
        } else if operation == "sub" {
            rt := numberone - numbertwo
            fmt.Println("Result ", rt)
        } else if operation == "mul" {
            rt := numberone * numbertwo
            fmt.Println("Result ", rt)
        } else if operation == "sqrt" {
            rt := math.Sqrt(numberone)
            fmt.Println("Result ", rt)
        } else {
            help()
        }
        return nil
    }
    //第5步
    app.Run(os.Args)
